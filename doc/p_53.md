## 5.3 Compiler un programme qui fait un putchar()
Command for the test
```bash
./chef.py -vv test backend -bw tests/grammar/valid/p_53.c -O
```

Result :
```ASM
.file  "p_53.c"
.text

.Ltext0:
  .globl main
  .type main, @function


main:
  pushq   %rbp
  movq    %rsp, %rbp
  # TODO: Prolog
  subq $1024, %rsp

  jmp    .L1_innerbeginof_main

.L1_innerbeginof_main:
  movl    $79, %r10d
  pushq    %rdi
  movq    %r10, %rdi

  call    putchar
  popq     %rdi
  movl    $75, %r10d
  pushq    %rdi
  movq    %r10, %rdi

  call    putchar
  popq     %rdi
  movl    $10, %r10d
  pushq    %rdi
  movq    %r10, %rdi

  call    putchar
  popq     %rdi
  jmp    .L0_endof_main

.L0_endof_main:
  leave
  ret
```
