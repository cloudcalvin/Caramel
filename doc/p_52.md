```ASM
.file  "p_52.c"
.text

.Ltext0:
  .globl main
  .type main, @function


main:
  pushq   %rbp
  movq    %rsp, %rbp
  # TODO: Prolog
  subq $1024, %rsp

  jmp    .L1_innerbeginof_main

.L1_innerbeginof_main:
  movl    $0, %r10d
  movq    %r10, -8(%rbp) # x
  movl    $0, %r10d
  movq    %r10, -16(%rbp) # y
  movl    $0, %r10d
  movq    %r10, -24(%rbp) # z
  jmp    .L0_endof_main

.L0_endof_main:
  leave
  ret
```

We did not manage a valid stack size for now. So we add 1024 bytes to the stack for the function context.
